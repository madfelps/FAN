67 void QMeter::drawScale(QPainter *painter)
68 {
69  painter->save();
70  painter->setPen(m_foreground);
71  double startRad=m_startAngle*PI/180.0;
72  startRad+=PI/2;
73  double deltaRad=(m_endAngle-m_startAngle)/(m_steps)*(PI/180);
74  double sina,cosa;
75  for ( int i=0; i<=m_steps; i++ )
76  {
77  sina=sin((double)(startRad+i*deltaRad));
78  cosa=cos((double)(startRad+i*deltaRad));
79  double tmpVal=i*((m_maxValue-m_minValue)/m_steps);
80  tmpVal+=m_minValue;
81  QString str = QString( "%1" ).arg(tmpVal,0,'f',m_precision);
82  QFontMetricsF fm( this->font() );

84  double h=fm.size(Qt::TextSingleLine,str).height();
85  int x=(int)((38*sina)-(w/2));
86  int y=(int)((38*cosa)+(h/4));
