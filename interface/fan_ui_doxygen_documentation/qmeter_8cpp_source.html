229 void QMeter::thresholdManager()
230 {
231 
232  if(m_value > m_threshold_medium && m_value < m_threshold && !m_thresholdMediumFlag)
233  {
234  m_thresholdMediumFlag=true;
235 
236  }
237  else if(m_value > m_threshold && !m_thresholdFlag)
238  {
239  m_thresholdFlag=true;
240  m_thresholdMediumFlag = false;
241  emit thresholdAlarm(m_thresholdFlag);
242  }
243  else if(m_value < m_threshold && m_thresholdFlag)
244  {
245  m_thresholdFlag=false;
246  m_thresholdMediumFlag = false;
247  emit thresholdAlarm(m_thresholdFlag);
248  }
249  else if(m_value < m_threshold_medium && m_thresholdMediumFlag)
250  {
251  m_thresholdMediumFlag = false;
252  }
253 }
