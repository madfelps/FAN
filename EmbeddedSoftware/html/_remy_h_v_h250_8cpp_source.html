62 int CutBytes::CutByteInterval(unsigned char* CAN_DATA, int CanDataPosition, int MS_Position, int LS_Position, int CuttedByte){
63  CuttedByte = CAN_DATA[CanDataPosition]; //verificar a questão da tipagem
64 
65  if(MS_Position == 7){
66  CuttedByte = CuttedByte & 0b11111111;
67  }
68  if(MS_Position == 6){
69  CuttedByte = CuttedByte & 0b01111111;
70  }
71  if(MS_Position == 5){
72  CuttedByte = CuttedByte & 0b00111111;
73  }
74  if(MS_Position == 4){
75  CuttedByte = CuttedByte & 0b00011111;
76  }
77  if(MS_Position == 3){
78  CuttedByte = CuttedByte & 0b00001111;
79  }
80  if(MS_Position == 2){
81  CuttedByte = CuttedByte & 0b00000111;
82  }
83  if(MS_Position == 1){
84  CuttedByte = CuttedByte & 0b00000011;
85  }
86  if(MS_Position == 0){
87  CuttedByte = CuttedByte & 0b00000001;
88  }
89  CuttedByte = (CuttedByte >> LS_Position);
90  return CuttedByte;
